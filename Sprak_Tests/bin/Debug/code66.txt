
void add(array a, string s)
	number i = indexOfLastElement(a)
	a[i + 1] = s
end

number indexOfLastElement(array a)
	array indexes = GetIndexes(a)
	return indexes[Count(indexes) - 1]
end

void MyRemove(array a, string s)
	loop i in GetIndexes(a)
		if(a[i] == s)
			Remove(a, i)
		end
	end
end
